        :param roughness: an estimation of the surface roughness
        :param path: the path for the datafile


        :type voidData: int
        :type geoTrnasform: tuple
        :type spatialReference: ??
        :type dataOrigin: str
        :type dataFeature: str
        :type accuracy: float
        :type roughness: float
        :type path: str
        """

        super(RasterProperty, self).__init__(points)

        self.__rasterData = RasterData(data, gridSpacing, **kwargs)
